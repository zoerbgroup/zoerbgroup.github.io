    function kw(H) {

      return 1e-14 / H;

    }


    function buildKas(pKas) {

      return pKas.map(pKa => Math.pow(10, -pKa));

    }


    function speciesFractions(H, kas) {

      const n = kas.length;

      const terms = new Array(n + 1).fill(0);

      terms[0] = 1;

      for (let i = 1; i <= n; i++) {

        terms[i] = terms[i - 1] * kas[i - 1] / H;

      }

      const denom = terms.reduce((sum, val) => sum + val, 0);

      return terms.map(term => term / denom);

    }


    function averageDeprotonation(alpha) {

      let total = 0;

      for (let i = 0; i < alpha.length; i++) total += i * alpha[i];

      return total;

    }


    function chargeBalanceResidual(logH, params) {

      const { Ca, VaL, Cb, VbL, kas } = params;

      const H = Math.pow(10, logH);

      const Vtotal = VaL + VbL;

      const cAcidFormal = (Ca * VaL) / Vtotal;

      const cNa = (Cb * VbL) / Vtotal;

      const alpha = speciesFractions(H, kas);

      const avgNegCharge = averageDeprotonation(alpha);

      const acidNegativeCharge = cAcidFormal * avgNegCharge;

      return H + cNa - kw(H) - acidNegativeCharge;

    }


    function solvePH(params) {

      let lo = -14;

      let hi = 0;

      let flo = chargeBalanceResidual(lo, params);

      let fhi = chargeBalanceResidual(hi, params);


      if (Math.abs(flo) < 1e-18) return 14;

      if (Math.abs(fhi) < 1e-18) return 0;


      if (flo * fhi > 0) {

        const samples = 400;

        let prevX = lo;

        let prevF = flo;

        let found = false;


        for (let i = 1; i <= samples; i++) {

          const x = lo + (hi - lo) * i / samples;

          const fx = chargeBalanceResidual(x, params);

          if (prevF === 0) return clamp(-prevX, 0, 14);

          if (prevF * fx <= 0) {

            lo = prevX;

            hi = x;

            flo = prevF;

            fhi = fx;

            found = true;

            break;

          }

          prevX = x;

          prevF = fx;

        }


        if (!found) return flo > 0 ? 0 : 14;

      }


      for (let i = 0; i < 100; i++) {

        const mid = 0.5 * (lo + hi);

        const fm = chargeBalanceResidual(mid, params);

        if (Math.abs(fm) < 1e-16 || Math.abs(hi - lo) < 1e-12) return clamp(-mid, 0, 14);

        if (flo * fm <= 0) {

          hi = mid;

          fhi = fm;

        } else {

          lo = mid;

          flo = fm;

        }

      }


      return clamp(-0.5 * (lo + hi), 0, 14);

    }


    function computePH(VbL, params) {

      return solvePH({ ...params, VbL });

    }


    function interpolatePHFromData(volumeMl) {

      const data = currentPlot.data;

      if (!data.length) return null;

      const maxIndex = data.length - 1;

      const clampedVolume = clamp(volumeMl, 0, currentPlot.maxX);

      const indexFloat = clampedVolume / FIXED_STEP_ML;

      const lowerIndex = Math.floor(indexFloat);

      const upperIndex = Math.min(maxIndex, Math.ceil(indexFloat));

      const lower = data[lowerIndex];

      const upper = data[upperIndex];

      if (!lower || !upper) return null;

      if (lowerIndex === upperIndex) return lower.pH;

      const t = (clampedVolume - lower.volumeMl) / (upper.volumeMl - lower.volumeMl);

      return lower.pH + t * (upper.pH - lower.pH);

    }


    function solveVolumeForPH(targetPH) {

      const params = getParams();

      const lowerPH = computePH(0, params);

      const upperPH = computePH(currentPlot.maxX / 1000, params);

      const minPH = Math.min(lowerPH, upperPH);

      const maxPH = Math.max(lowerPH, upperPH);

      if (targetPH < minPH || targetPH > maxPH) return null;


      let lo = 0;

      let hi = currentPlot.maxX / 1000;

      let flo = computePH(lo, params) - targetPH;

      let fhi = computePH(hi, params) - targetPH;


      if (Math.abs(flo) < 1e-10) return 0;

      if (Math.abs(fhi) < 1e-10) return currentPlot.maxX;


      for (let i = 0; i < 80; i++) {

        const mid = 0.5 * (lo + hi);

        const fm = computePH(mid, params) - targetPH;

        if (Math.abs(fm) < 1e-8 || Math.abs(hi - lo) < 1e-8) {

          return mid * 1000;

        }

        if (flo * fm <= 0) {

          hi = mid;

          fhi = fm;

        } else {

          lo = mid;

          flo = fm;

        }

      }


      return 0.5 * (lo + hi) * 1000;

    }


    function getSpeciesName(n, i) {

      if (i === 0) return `H${n}A`;

      if (i === n) return `A${n === 1 ? '' : i + '-'} `;

      const hCount = n - i;

      const charge = i;

      const acidCore = hCount > 0 ? `H${hCount}A` : 'A';

      return charge === 0 ? acidCore : `${acidCore}${charge === 1 ? '-' : charge + '-'}`;

    }


    function getSpeciesDisplayName(n, i) {

      if (i === 0) {

        return `H<sub>${n}</sub>A`;

      }

      const remainingH = n - i;

      const base = remainingH > 0 ? `H<sub>${remainingH}</sub>A` : 'A';

      if (i === 1) return `${base}<sup>−</sup>`;

      return `${base}<sup>${i}−</sup>`;

    }


    function getParams() {

      const n = getSelectedProtons();

      const Ca = parseFloat(acidConcEl.value);

      const VaL = parseFloat(acidVolEl.value) / 1000;

      const Cb = parseFloat(baseConcEl.value);

      const pKas = [

        parseFloat(pKa1El.value),

        parseFloat(pKa2El.value),

        parseFloat(pKa3El.value),

        parseFloat(pKa4El.value),

        parseFloat(pKa5El.value)

      ].slice(0, n).sort((a, b) => a - b);

      const kas = buildKas(pKas);

      return { n, Ca, VaL, Cb, pKas, kas };

    }


    function simulate() {

      const params = getParams();

      const { n, Ca, VaL, Cb, kas } = params;

      const acidMoles = Ca * VaL;

      const equivalenceVolumes = Array.from({ length: n }, (_, i) => ((i + 1) * acidMoles / Cb) * 1000);

      const maxEq = equivalenceVolumes[equivalenceVolumes.length - 1] || 0;

      const maxBaseVolMl = maxEq + 50;

      const data = [];

      const speciesData = [];

      const derivativeData = [];

      const speciesNames = Array.from({ length: n + 1 }, (_, i) => getSpeciesDisplayName(n, i));


      for (let volumeMl = 0; volumeMl <= maxBaseVolMl + 1e-12; volumeMl += FIXED_STEP_ML) {

        const VbL = volumeMl / 1000;

        const pH = computePH(VbL, params);

        data.push({ volumeMl, pH });

      }


      for (let i = 0; i < data.length; i++) {

        let derivative;

        if (i === 0) {

          derivative = (data[i + 1].pH - data[i].pH) / (data[i + 1].volumeMl - data[i].volumeMl);

        } else if (i === data.length - 1) {

          derivative = (data[i].pH - data[i - 1].pH) / (data[i].volumeMl - data[i - 1].volumeMl);

        } else {

          derivative = (data[i + 1].pH - data[i - 1].pH) / (data[i + 1].volumeMl - data[i - 1].volumeMl);

        }

        derivativeData.push({ volumeMl: data[i].volumeMl, dpHdV: derivative });

      }


      const speciesPHStep = 0.02;

      for (let pH = 0; pH <= 14 + 1e-12; pH += speciesPHStep) {

        const H = Math.pow(10, -pH);

        const alpha = speciesFractions(H, kas);

        speciesData.push({ pH, fractions: alpha });

      }


      currentPlot.data = data;

      currentPlot.eqVolumes = equivalenceVolumes;

      currentPlot.maxX = maxBaseVolMl;

      currentPlot.selectedPoint = null;

      currentPlot.hoverPoint = null;

      currentPlot.speciesData = speciesData;

      currentPlot.speciesHoverPoint = null;

      currentPlot.speciesNames = speciesNames;

      currentPlot.derivativeData = derivativeData;


      lookupVolumeInputEl.max = format(maxBaseVolMl, 2);

      buildSpeciesLegend(speciesNames);

      updateSelectedPointText();

      updateHoverText();

      drawChart(data, equivalenceVolumes, maxBaseVolMl);

      drawSpeciesChart(speciesData, speciesNames);

      runLookup();

    }

    function niceUpperFive(value) {

      return Math.ceil(value / 5) * 5;

    }


    function buildScales(targetCanvas, maxX, minY, maxY, bottomPad) {

      const w = targetCanvas.width;

      const h = targetCanvas.height;

      const pad = { left: 62, right: 20, top: 24, bottom: bottomPad };

      const plotW = w - pad.left - pad.right;

      const plotH = h - pad.top - pad.bottom;

      return {

        w,

        h,

        pad,

        plotW,

        plotH,

        minY,

        maxY,

        xScale(x) {

          return pad.left + (x / maxX) * plotW;

        },

        yScale(y) {

          return pad.top + plotH - ((y - minY) / (maxY - minY)) * plotH;

        }

      };

    }


    function drawAxes(targetCtx, scales, displayedMaxX, yTicks, yLabel, xLabel) {

      const { w, h, pad, xScale, yScale } = scales;


      targetCtx.strokeStyle = 'rgba(148,163,184,0.18)';

      targetCtx.lineWidth = 1;

      targetCtx.font = '12px Arial';

      targetCtx.fillStyle = '#cbd5e1';


      yTicks.forEach(tick => {

        const py = yScale(tick.value);

        targetCtx.beginPath();

        targetCtx.moveTo(pad.left, py);

        targetCtx.lineTo(w - pad.right, py);

        targetCtx.stroke();

        targetCtx.fillText(tick.label, 28, py + 4);

      });


      // limit to max 20 ticks, all multiples of 5

      const baseStep = 5;

      const maxTicks = 20;

      const stepMultiplier = Math.max(1, Math.ceil(displayedMaxX / (baseStep * maxTicks)));

      const xTickStep = baseStep * stepMultiplier;


      for (let xv = 0; xv <= displayedMaxX + 1e-9; xv += xTickStep) {

        const px = xScale(xv);

        targetCtx.beginPath();

        targetCtx.moveTo(px, pad.top);

        targetCtx.lineTo(px, h - pad.bottom);

        targetCtx.stroke();

        const label = xv.toFixed(0);

        targetCtx.fillText(label, px - targetCtx.measureText(label).width / 2, h - pad.bottom + 16);

      }


      targetCtx.strokeStyle = '#94a3b8';

      targetCtx.lineWidth = 1.5;

      targetCtx.beginPath();

      targetCtx.moveTo(pad.left, pad.top);

      targetCtx.lineTo(pad.left, h - pad.bottom);

      targetCtx.lineTo(w - pad.right, h - pad.bottom);

      targetCtx.stroke();


      targetCtx.save();

      targetCtx.translate(18, h / 2);

      targetCtx.rotate(-Math.PI / 2);

      targetCtx.fillStyle = '#e5e7eb';

      targetCtx.font = '14px Arial';

      targetCtx.fillText(yLabel, 0, 0);

      targetCtx.restore();


      targetCtx.fillStyle = '#e5e7eb';

      targetCtx.font = '14px Arial';

      // place x-axis label slightly above the bottom, below tick labels without overlap

      targetCtx.fillText(xLabel, w / 2 - targetCtx.measureText(xLabel).width / 2, h - pad.bottom + 32);

    }


    function drawCrosshair(targetCtx, scales, xValue, yValue, dotColor) {

      const { w, h, pad, xScale, yScale } = scales;

      const sx = xScale(xValue);

      const sy = yScale(yValue);

      targetCtx.strokeStyle = 'rgba(255, 255, 255, 0.65)';

      targetCtx.lineWidth = 1;

      targetCtx.setLineDash([4, 4]);

      targetCtx.beginPath();

      targetCtx.moveTo(sx, pad.top);

      targetCtx.lineTo(sx, h - pad.bottom);

      targetCtx.moveTo(pad.left, sy);

      targetCtx.lineTo(w - pad.right, sy);

      targetCtx.stroke();

      targetCtx.setLineDash([]);

      targetCtx.fillStyle = dotColor;

      targetCtx.beginPath();

      targetCtx.arc(sx, sy, 4, 0, Math.PI * 2);

      targetCtx.fill();

    }


    function drawChart(data, eqVolumes, maxX) {

      const displayedMaxX = niceUpperFive(maxX);

      const scales = buildScales(canvas, displayedMaxX, 0, 14, 64);

      currentPlot.scales = scales;

      const { w, h, pad, xScale, yScale } = scales;


      ctx.clearRect(0, 0, w, h);

      ctx.fillStyle = '#020617';

      ctx.fillRect(0, 0, w, h);


      drawAxes(

        ctx,

        scales,

        displayedMaxX,

        Array.from({ length: 8 }, (_, i) => ({ value: i * 2, label: String(i * 2) })),

        'pH',

        'Base added (mL)'

      );


      if (isDerivativeOverlayOn() && currentPlot.derivativeData.length) {

        const derivValues = currentPlot.derivativeData.map(pt => pt.dpHdV).filter(v => Number.isFinite(v));

        const derivMin = Math.min(...derivValues);

        const derivMax = Math.max(...derivValues);

        const safeMin = Math.min(0, derivMin);

        const safeMax = derivMax === safeMin ? safeMin + 1 : derivMax;

        const derivScale = value => pad.top + (h - pad.top - pad.bottom) - ((value - safeMin) / (safeMax - safeMin)) * (h - pad.top - pad.bottom);


        ctx.save();

        ctx.strokeStyle = 'rgba(167, 139, 250, 0.95)';

        ctx.lineWidth = 2;

        ctx.setLineDash([8, 5]);

        ctx.beginPath();

        currentPlot.derivativeData.forEach((pt, i) => {

          const x = xScale(pt.volumeMl);

          const y = derivScale(pt.dpHdV);

          if (i === 0) ctx.moveTo(x, y);

          else ctx.lineTo(x, y);

        });

        ctx.stroke();

        ctx.setLineDash([]);


        ctx.fillStyle = 'rgba(167, 139, 250, 0.95)';

        ctx.font = '12px Arial';

        const label = 'dpH/dV';

        ctx.fillText(label, w - pad.right - ctx.measureText(label).width - 4, pad.top + 14);

        ctx.restore();

      }


      ctx.strokeStyle = '#22c55e';

      ctx.lineWidth = 3;

      ctx.beginPath();

      data.forEach((pt, i) => {

        const x = xScale(pt.volumeMl);

        const y = yScale(pt.pH);

        if (i === 0) ctx.moveTo(x, y);

        else ctx.lineTo(x, y);

      });

      ctx.stroke();


      const last = data[data.length - 1];

      ctx.fillStyle = '#22c55e';

      ctx.beginPath();

      ctx.arc(xScale(last.volumeMl), yScale(last.pH), 4, 0, Math.PI * 2);

      ctx.fill();


      if (isLiveHoverOn() && currentPlot.hoverPoint) {

        drawCrosshair(ctx, scales, currentPlot.hoverPoint.volumeMl, currentPlot.hoverPoint.pH, '#ffffff');

      }


      if (currentPlot.selectedPoint) {

        const sx = xScale(currentPlot.selectedPoint.volumeMl);

        const sy = yScale(currentPlot.selectedPoint.pH);

        ctx.strokeStyle = 'rgba(245, 158, 11, 0.85)';

        ctx.lineWidth = 1.25;

        ctx.setLineDash([4, 4]);

        ctx.beginPath();

        ctx.moveTo(sx, pad.top);

        ctx.lineTo(sx, h - pad.bottom);

        ctx.moveTo(pad.left, sy);

        ctx.lineTo(w - pad.right, sy);

        ctx.stroke();

        ctx.setLineDash([]);


        ctx.fillStyle = '#f59e0b';

        ctx.beginPath();

        ctx.arc(sx, sy, 5, 0, Math.PI * 2);

        ctx.fill();

      }

    }

    function runLookup() {

      if (!currentPlot.data.length) {

        lookupResultTextEl.textContent = 'Run the simulation first.';

        return;

      }


      const mode = lookupModeEl.value;

      if (mode === 'ph_from_volume') {

        const requestedVolume = parseFloat(lookupVolumeInputEl.value);

        if (!Number.isFinite(requestedVolume)) {

          lookupResultTextEl.textContent = 'Enter a valid base volume.';

          return;

        }

        const volumeMl = clamp(requestedVolume, 0, currentPlot.maxX);

        const pH = interpolatePHFromData(volumeMl);

        lookupResultTextEl.textContent = `At x = ${format(volumeMl, 2)} mL, the titration curve gives pH = ${format(pH, 4)}.`;

      } else {

        const targetPH = parseFloat(lookupPHInputEl.value);

        if (!Number.isFinite(targetPH) || targetPH < 0 || targetPH > 14) {

          lookupResultTextEl.textContent = 'Enter a pH between 0 and 14.';

          return;

        }

        const volumeMl = solveVolumeForPH(targetPH);

        if (volumeMl === null) {

          lookupResultTextEl.textContent = `pH ${format(targetPH, 2)} is outside the simulated titration-curve range.`;

          return;

        }

        lookupResultTextEl.textContent = `For pH = ${format(targetPH, 2)}, the titration curve reaches x = ${format(volumeMl, 3)} mL of added base.`;

      }

    }

    function nearestPointByCanvasX(event, targetCanvas, plotData, scales, xKey = 'volumeMl') {

      if (!plotData.length || !scales) return null;

      const rect = targetCanvas.getBoundingClientRect();

      const scaleXFactor = targetCanvas.width / rect.width;

      const clickX = (event.clientX - rect.left) * scaleXFactor;


      let bestPoint = null;

      let bestDistance = Infinity;

      plotData.forEach(pt => {

        const px = scales.xScale(pt[xKey]);

        const dx = Math.abs(px - clickX);

        if (dx < bestDistance) {

          bestDistance = dx;

          bestPoint = pt;

        }

      });

      return bestPoint;

    }
